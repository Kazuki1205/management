public class Production implements Serializable{

    private long id; //ID
    private Report[] reports; //日報クラスのリスト
    private Item item; //商品クラス
    private String lotNumber; //製作番号
    private int lotQuantity; //製作数
    private String scheduledCompletionDate; //完成予定日
    private String completionDate; //完成日
    private int departmentCompletionQuantityTotal; //入力者の所属する部署毎の、該当製番に対する完了数計
    private int failureQuantityTotal; //該当製番に対する不良数計
    private int storingQuantityTotal; //該当製番に対する入庫数計
    private String createdAt; //登録日時
    private String updatedAt; //更新日時
    private short invalid; //削除フラグ

    public long getId(){
        return id;
    }

    public void setId(long id){
        this.id = id;
    }

    public Report[] getReports(){
        return reports;
    }

    public void setReports(Report[] reports){
        this.reports = reports;
    }

    public Item getItem(){
        return item;
    }

    public void setItem(Item item){
        this.item = item;
    }

    public String getLotNumber(){
        return lotNumber;
    }

    public void setLotNumber(String lotNumber){
        this.lotNumber = lotNumber;
    }

    public int getLotQuantity(){
        return lotQuantity;
    }

    public void setLotQuantity(int lotQuantity){
        this.lotQuantity = lotQuantity;
    }

    public String getScheduledCompletionDate(){
        return scheduledCompletionDate;
    }

    public void setScheduledCompletionDate(String scheduledCompletionDate){
        this.scheduledCompletionDate = scheduledCompletionDate;
    }

    public String getCompletionDate(){
        return completionDate;
    }

    public void setCompletionDate(String completionDate){
        this.completionDate = completionDate;
    }

    public int getDepartmentCompletionQuantityTotal(){
        return departmentCompletionQuantityTotal;
    }

    public void setDepartmentCompletionQuantityTotal(int departmentCompletionQuantityTotal){
        this.departmentCompletionQuantityTotal = departmentCompletionQuantityTotal;
    }

    public int getFailureQuantityTotal(){
        return failureQuantityTotal;
    }

    public void setFailureQuantityTotal(int failureQuantityTotal){
        this.failureQuantityTotal = failureQuantityTotal;
    }

    public int getStoringQuantityTotal(){
        return storingQuantityTotal;
    }

    public void setStoringQuantityTotal(int storingQuantityTotal){
        this.storingQuantityTotal = storingQuantityTotal;
    }

    public String getCreatedAt(){
        return createdAt;
    }

    public void setCreatedAt(String createdAt){
        this.createdAt = createdAt;
    }

    public String getUpdatedAt(){
        return updatedAt;
    }

    public void setUpdatedAt(String updatedAt){
        this.updatedAt = updatedAt;
    }

    public short getInvalid(){
        return invalid;
    }

    public void setInvalid(short invalid){
        this.invalid = invalid;
    }

}
