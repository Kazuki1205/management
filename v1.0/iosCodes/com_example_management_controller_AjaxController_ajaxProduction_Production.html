#import <Foundation/Foundation>;

@interface Production : NSObject

//ID
@property (nonatomic, strong) long id;
//日報クラスのリスト
@property (nonatomic, strong) Report[] reports;
//商品クラス
@property (nonatomic, strong) Item item;
//製作番号
@property (nonatomic, assign) NSString lotNumber;
//製作数
@property (nonatomic, strong) NSInteger lotQuantity;
//完成予定日
@property (nonatomic, assign) NSString scheduledCompletionDate;
//完成日
@property (nonatomic, assign) NSString completionDate;
//入力者の所属する部署毎の、該当製番に対する完了数計
@property (nonatomic, strong) NSInteger departmentCompletionQuantityTotal;
//該当製番に対する不良数計
@property (nonatomic, strong) NSInteger failureQuantityTotal;
//該当製番に対する入庫数計
@property (nonatomic, strong) NSInteger storingQuantityTotal;
//登録日時
@property (nonatomic, assign) NSString createdAt;
//更新日時
@property (nonatomic, assign) NSString updatedAt;
//削除フラグ
@property (nonatomic, strong) short invalid;


@end



#import <Foundation/Foundation>;

@interface Report : NSObject

//ID
@property (nonatomic, strong) long id;
//製作ID
@property (nonatomic, strong) long productionId;
//部署ID
@property (nonatomic, strong) long departmentId;
//製作クラス
@property (nonatomic, strong) Production production;
//社員履歴クラス
@property (nonatomic, strong) EmployeeHistory employeeHistory;
//完了数
@property (nonatomic, strong) NSInteger completionQuantity;
//不良数
@property (nonatomic, strong) NSInteger failureQuantity;
//部署毎の、該当製番に対する完了数計
@property (nonatomic, strong) NSInteger departmentCompletionQuantityTotal;
//部署毎の、該当製番に対する不良数計
@property (nonatomic, strong) NSInteger departmentFailureQuantityTotal;
//登録日時
@property (nonatomic, assign) NSString createdAt;
//更新日時
@property (nonatomic, assign) NSString updatedAt;
//最終更新日時
@property (nonatomic, assign) NSString updatedAtLatest;
//削除フラグ
@property (nonatomic, strong) short invalid;


@end



#import <Foundation/Foundation>;
